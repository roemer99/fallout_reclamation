improve_our_cultural_view = {
	type = owncountry

    ai_tick = never

	potential = {
		scope:actor = {
			culture = {
				dominant_country ?= scope:actor
			}
		}
	}

	allow = {
	}

    price = price:improve_our_cultural_view_price

	select_trigger = {
		looking_for_a = culture
		target_flag = target
		name = "improve_our_cultural_view_choose_culture"
		column = {
			data = name
		}
		visible = {
			not = { this = scope:actor.culture }
		}
        enabled = {
			scope:actor = {
				culture = {
					NOT = {
						cultural_view = {
							target = scope:target
							value = kindred
						}
					}
				}
			}
		}
	}

	effect = {
		scope:actor = {
			culture = {
				change_cultural_view = {
					target = scope:target
					change = 1
				}
			}
		}
	}
}
